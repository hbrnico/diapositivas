Beecrowd 1173: Array Fill I

Leer un número y ponerlo en la primera posición de un arreglo N[10]. En cada posición subsecuente, colocar el doble del anterior.

ENTRADA

1

SALIDA ESPERADA

N[0] = 1
N[1] = 2
N[2] = 4
...

Solución: Array Fill I


#include <iostream>
#define TAM 10
using namespace std;

void inicializar_teclado(int [], int);
void imprimir_vector(int [], int);

int main() {
    int vector[TAM];
    inicializar_teclado(vector, TAM);
    imprimir_vector(vector, TAM);
    return 0;
}

void inicializar_teclado(int V[], int tam){
    cin >> V[0];
    for(int i=1; i<tam; i++)
        V[i] = 2 * V[i-1];
}
    
void imprimir_vector(int V[], int tam){
    for(int i=0; i<tam; i++)
        cout << "N[" << i << "] = " << V[i] << endl;
}

Beecrowd 1171: Frecuencias

Leer N números enteros positivos (hasta el valor 2000) e imprimir cuántas veces aparece cada uno, ordenados de forma ascendente.

Estrategia Clave

En lugar de buscar y contar número por número (que sería muy lento), usamos un arreglo de frecuencias donde el índice representa el número leído, y el valor es la cantidad de apariciones.

Solución: Frecuencias


#define TAMFRECUENCIAS 20000
// Inicializamos el arreglo de frecuencias todo en 0 con {}
// Ej: frecuencias[5] guardará cuántas veces leímos el "5"

int main() {
    int frecuencias[TAMFRECUENCIAS] {};
    int n;
    cin >> n;
    
    ingresaElementos(frecuencias, n);
    imprimeFrecuencias(frecuencias);
    return 0;
}

void ingresaElementos(int frecuencias[], int n) {
    for (int i = 0; i < n; i++) {
        int ingreso;
        cin >> ingreso;
        frecuencias[ingreso]++; // Magia: Acumulamos directamente en el índice!
    }
}

void imprimeFrecuencias(int frecuencias[]) {
    for (int i = 0; i < TAMFRECUENCIAS; i++) {
        if (frecuencias[i] != 0) 
            cout << i << " aparece " << frecuencias[i] << " vez(es)" << endl;
    }
}

Ejercicio Avanzado

Se leen 10 valores enteros menores que 30. Luego se leen valores hasta que la suma de los dígitos de uno leído, sea igual a alguno de los 10 valores inicialmente leídos. Informar el número.

Ejemplo:

  • Arreglo Inicial: 1 6 25 18 23 2 6 19 14 13
  • Se lee 9245 -> Suma = 20 (No está)
  • Se lee 340 -> Suma = 7 (No está)
  • Se lee 694 -> Suma = 19 (¡Sí está en el arreglo!). Termina e imprime 694.

Resolución Modularizada


int main() {
    int iniciales[10], ingreso;
    inicializaArreglo(iniciales, 10);
    
    do {
        cout << "Ingrese un numero:" << endl;
        cin >> ingreso;
    } while(!existeNumero(iniciales, 10, sumaDigitos(ingreso)));
    
    cout << "Cumple condicion: " << ingreso << endl;
    return 0;
}

// Búsqueda secuencial estándar
bool existeNumero(int V[], int tam, int buscado) {
    bool encontrado = false;
    int i = 0;
    while (i < tam && !encontrado) {
        if (V[i] == buscado) encontrado = true;
        i++;
    }
    return encontrado;
}

int sumaDigitos(int numero) {
    if (numero < 10) return numero;
    return numero % 10 + sumaDigitos(numero/10); // Recursividad elegante
}
