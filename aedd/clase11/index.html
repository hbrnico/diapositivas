omegaUp 17818: El doble del término anterior

Leer un número y ponerlo en la primera posición de un arreglo N[10]. En cada posición subsecuente, colocar el doble del anterior.

ENTRADA

1

SALIDA ESPERADA

N[0] = 1
N[1] = 2
N[2] = 4
...

Solución: El doble del término anterior


#include <iostream>
#define TAM 10
using namespace std;

void inicializar_teclado(int [], int);
void imprimir_vector(int [], int);

int main() {
    int vector[TAM];
    inicializar_teclado(vector, TAM);
    imprimir_vector(vector, TAM);
    return 0;
}

void inicializar_teclado(int V[], int tam){
    cin >> V[0];
    for(int i=1; i<tam; i++)
        V[i] = 2 * V[i-1];
}
    
void imprimir_vector(int V[], int tam){
    for(int i=0; i<tam; i++)
        cout << "N[" << i << "] = " << V[i] << endl;
}

omegaUp 11232: Frecuencia de una Letra

Leer una palabra p y una letra l. Informar cuántas veces aparece la letra en la palabra.

ENTRADA

panama a

SALIDA ESPERADA

3

Estrategia Clave

Usamos un arreglo de frecuencias (de tamaño 256) donde el índice representa el valor ASCII del carácter, y el valor es su cantidad de apariciones.

Solución: Frecuencia de una Letra


#include <iostream>
#include <string>
using namespace std;

void contarFrecuencias(string, int []);
int obtenerFrecuencia(int [], char);

int main() {
    int frecuencias[256] {}; // Inicializa todo en 0
    string p;
    char l;
    
    if (cin >> p >> l) {
        contarFrecuencias(p, frecuencias);
        cout << obtenerFrecuencia(frecuencias, l) << endl;
    }
    return 0;
}

void contarFrecuencias(string palabra, int frecuencias[]) {
    for (int i = 0; i < palabra.length(); i++) {
        char caracter = palabra[i];
        frecuencias[caracter]++; // Magia: Acumulamos directamente en el índice ASCII!
    }
}

int obtenerFrecuencia(int frecuencias[], char letra) {
    return frecuencias[letra];
}

Ejercicio Avanzado

Se leen 10 valores enteros menores que 30. Luego se leen valores hasta que la suma de los dígitos de uno leído, sea igual a alguno de los 10 valores inicialmente leídos. Informar el número.

Ejemplo:

  • Arreglo Inicial: 1 6 25 18 23 2 6 19 14 13
  • Se lee 9245 -> Suma = 20 (No está)
  • Se lee 340 -> Suma = 7 (No está)
  • Se lee 694 -> Suma = 19 (¡Sí está en el arreglo!). Termina e imprime 694.

Resolución Modularizada


int main() {
    int iniciales[10], ingreso;
    inicializaArreglo(iniciales, 10);
    
    do {
        cout << "Ingrese un numero:" << endl;
        cin >> ingreso;
    } while(!existeNumero(iniciales, 10, sumaDigitos(ingreso)));
    
    cout << "Cumple condicion: " << ingreso << endl;
    return 0;
}

// Búsqueda secuencial estándar
bool existeNumero(int V[], int tam, int buscado) {
    bool encontrado = false;
    int i = 0;
    while (i < tam && !encontrado) {
        if (V[i] == buscado) encontrado = true;
        i++;
    }
    return encontrado;
}

int sumaDigitos(int numero) {
    if (numero < 10) return numero;
    return numero % 10 + sumaDigitos(numero/10); // Recursividad elegante
}
