Ejercicio 5 - Guía 4

Descripción: Se solicita escribir el código para ingresar desde el teclado un número de día de la semana (1-7) y un número de mes (1-12), y que imprima el nombre del día de la semana y el del mes correspondientes.

Si el día o el mes ingresado es incorrecto deberá mostrarse un mensaje de error.


Ejemplo:

  • Ingresando 3, 8 → Martes y Agosto
  • Ingresando 12, 7 → Día inexistente.

Solución: Ejercicio 5


#include <iostream>
using namespace std;

int main() {
    int dia, mes;
    
    cout << "Dia de la semana (1-7): "; cin >> dia;
    cout << "Mes (1-12): "; cin >> mes;
    
    if (dia > 7 || dia < 1 || mes > 12 || mes < 1) {
        cout << "Dia inexistente" << endl;
    } else {
        cout << "Dia: ";
        switch(dia) {
            case 1: cout << "Lunes"; break;
            case 2: cout << "Martes"; break;
            case 3: cout << "Miercoles"; break;
            case 4: cout << "Jueves"; break;
            case 5: cout << "Viernes"; break;
            case 6: cout << "Sabado"; break;
            case 7: cout << "Domingo"; break;
        }
        
        switch(mes) {
            case 1: cout << " y Enero\n"; break;
            case 2: cout << " y Febrero\n"; break;
            case 3: cout << " y Marzo\n"; break;
            case 4: cout << " y Abril\n"; break;
            case 5: cout << " y Mayo\n"; break;
            case 6: cout << " y Junio\n"; break;
            case 7: cout << " y Julio\n"; break;
            case 8: cout << " y Agosto\n"; break;
            case 9: cout << " y Septiembre\n"; break;
            case 10: cout << " y Octubre\n"; break;
            case 11: cout << " y Noviembre\n"; break;
            case 12: cout << " y Diciembre\n"; break;
        }
    }
    return 0;
}

Ejercicio 9 - Guía 4

Escribir un programa que dado un número del 0 al 9, escriba en pantalla los números restantes hasta el 10.


Ejemplo:

  • Ingrese un número: 7
  • 8 9 10

Solución: Ejercicio 9


#include <iostream>
using namespace std;

int main() {
    int numero;
    cout << "Ingrese un numero (0-9): ";
    cin >> numero;
    
    switch (numero) {
        case 0: cout << "1\t";
        case 1: cout << "2\t";
        case 2: cout << "3\t";
        case 3: cout << "4\t";
        case 4: cout << "5\t";
        case 5: cout << "6\t";
        case 6: cout << "7\t";
        case 7: cout << "8\t";
        case 8: cout << "9\t";
        case 9: cout << "10" << endl; break;
        default: cout << "Fuera del rango" << endl;
    }
    return 0;
}

Ejercicio 10 - Guía 4

Realizar un programa que permita calcular el perímetro de un triángulo. El programa mostrará un menú con las opciones Equilátero, Isósceles y Escaleno.

Si la opción es incorrecta, muestra un error. Si es correcta, solicitará los datos necesarios (deben ser positivos) y calculará el perímetro.

Menú:

  1. Equilatero
  2. Isósceles
  3. Escaleno

Solución: Ejercicio 10


#include <iostream>
using namespace std;

int main() {
    int tipo;
    float l1, l2, l3, p = 0;

    cout << "1- Equilatero\n2- Isoceles\n3- Escaleno\nTipo: ";
    cin >> tipo;

    if (tipo < 1 || tipo > 3) {
        cout << "Opcion invalida." << endl;
    } else {
        switch (tipo) {
            case 1:
                cout << "Lado: "; cin >> l1;
                if (l1 <= 0) cout << "El valor debe ser positivo." << endl;
                else p = l1 * 3;
                break;
            case 2:
                cout << "Lado repetido: "; cin >> l1;
                cout << "Lado distinto: "; cin >> l2;
                if (l1 <= 0 || l2 <= 0) cout << "Debe ser positivo." << endl;
                else p = l1 * 2 + l2;
                break;
            case 3:
                cout << "Lado 1: "; cin >> l1;
                cout << "Lado 2: "; cin >> l2;
                cout << "Lado 3: "; cin >> l3;
                if (l1 <= 0 || l2 <= 0 || l3 <= 0) cout << "Positivos." << endl;
                else p = l1 + l2 + l3;
                break;
        }
        
        if (p > 0) cout << "Perimetro: " << p << endl;
    }
    return 0;
}
